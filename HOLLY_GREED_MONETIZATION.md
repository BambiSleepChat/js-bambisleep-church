```dockerfile
COPY --from=build /app/dist ./dist
COPY --from=build /app/package*.json ./

# Set ownership
RUN chown -R bambisleep:bambisleep /app

# Switch to non-root user
USER bambisleep

# Environment variables (never hardcode secrets!)
ENV NODE_ENV=production
ENV PORT=3000

# Health check
HEALTHCHECK --interval=30s --timeout=3s --start-period=5s --retries=3 \
  CMD node dist/healthcheck.js || exit 1

# Expose port (documentation only)

# Start application
CMD ["node", "dist/main.js"]
```

## ðŸ’¾ Database Schema (PostgreSQL)

From `database-data-management.collection.yml` best practices:

### User & Subscription Tables

```sql
-- Users table (OWASP A07 compliant)
CREATE TABLE users (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  email VARCHAR(255) UNIQUE NOT NULL,
  password_hash VARCHAR(255) NOT NULL,  -- Argon2 hashed
  stripe_customer_id VARCHAR(255) UNIQUE,
  created_at TIMESTAMP DEFAULT NOW(),
  updated_at TIMESTAMP DEFAULT NOW(),
  last_login_at TIMESTAMP,
  failed_login_attempts INT DEFAULT 0,
  locked_until TIMESTAMP NULL
);

-- Indexes for performance
CREATE INDEX idx_users_email ON users(email);
CREATE INDEX idx_users_stripe_id ON users(stripe_customer_id);

-- Church Donations (subscriptions)
CREATE TABLE church_subscriptions (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id UUID REFERENCES users(id) ON DELETE CASCADE,
  stripe_subscription_id VARCHAR(255) UNIQUE NOT NULL,
  status VARCHAR(50) NOT NULL,  -- active, canceled, past_due
  current_period_start TIMESTAMP NOT NULL,
  current_period_end TIMESTAMP NOT NULL,
  cancel_at_period_end BOOLEAN DEFAULT FALSE,
  created_at TIMESTAMP DEFAULT NOW(),
  updated_at TIMESTAMP DEFAULT NOW()
);

CREATE INDEX idx_subscriptions_user ON church_subscriptions(user_id);
CREATE INDEX idx_subscriptions_stripe ON church_subscriptions(stripe_subscription_id);

-- Bambi Tokens (balance tracking)
CREATE TABLE token_balances (
  user_id UUID PRIMARY KEY REFERENCES users(id) ON DELETE CASCADE,
  balance INT NOT NULL DEFAULT 0 CHECK (balance >= 0),
  lifetime_earned INT NOT NULL DEFAULT 0,
  lifetime_spent INT NOT NULL DEFAULT 0,
  updated_at TIMESTAMP DEFAULT NOW()
);

-- Token transactions (audit trail)
CREATE TABLE token_transactions (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id UUID REFERENCES users(id) ON DELETE CASCADE,
  amount INT NOT NULL,  -- positive = credit, negative = debit
  transaction_type VARCHAR(50) NOT NULL,  -- purchase, spend, gift, reward
  description TEXT,
  stripe_payment_intent_id VARCHAR(255),
  created_at TIMESTAMP DEFAULT NOW()
);

CREATE INDEX idx_transactions_user ON token_transactions(user_id);
CREATE INDEX idx_transactions_type ON token_transactions(transaction_type);
CREATE INDEX idx_transactions_date ON token_transactions(created_at DESC);
```

**Security Features**:
- âœ… UUID primary keys (prevent enumeration)
- âœ… Foreign key constraints (referential integrity)
- âœ… CHECK constraints (data validation)
- âœ… Indexes for performance (EXPLAIN ANALYZE verified)
- âœ… ON DELETE CASCADE (GDPR compliance)
- âœ… Timestamps for audit trail

```javascript
import express from 'express';
import Stripe from 'stripe';
import { validators, validateRequest } from '../middleware/input-validation.js';
import { authLimiter, donationLimiter } from '../middleware/rate-limiting.js';
import { logger } from '../services/logger.js';

const router = express.Router();
const stripe = new Stripe(process.env.STRIPE_SECRET_KEY);

const CHURCH_DONATION_PRICE_ID = 'price_XXXXX';  // â‚¬5/month (prod_SoDd3WjNmvxWC9)

/**
 * POST /api/church/donate
 * Create Church Donation subscription checkout session
 */
router.post('/donate',
  authLimiter,
  donationLimiter,
  validators.email,
  validateRequest,
  async (req, res) => {
    try {
      const { email } = req.body;

      // Create or retrieve Stripe customer
      let customer;
      const existingCustomers = await stripe.customers.list({ email, limit: 1 });
      
      if (existingCustomers.data.length > 0) {
        customer = existingCustomers.data[0];
      } else {
        customer = await stripe.customers.create({
          email,
          metadata: {
            source: 'bambisleep-church',
            tier: 'church-patron'
          }
        });
      }

      // Create checkout session
      const session = await stripe.checkout.sessions.create({
        customer: customer.id,
        mode: 'subscription',
        payment_method_types: ['card'],
        line_items: [{
          price: CHURCH_DONATION_PRICE_ID,
          quantity: 1
        }],
        success_url: `${process.env.APP_URL}/church/success?session_id={CHECKOUT_SESSION_ID}`,
        cancel_url: `${process.env.APP_URL}/church/cancel`,
        subscription_data: {
          metadata: {
            tier: 'church-patron',
            perks: 'priority-support,exclusive-content,community-recognition'
          }
        },
        allow_promotion_codes: true,  // Enable promo codes
        billing_address_collection: 'required'
      });

      logger.info('Church donation session created', {
        customerId: customer.id,
        sessionId: session.id,
        email
      });

      res.json({
        sessionId: session.id,
        url: session.url
      });
    } catch (error) {
      logger.error('Church donation session creation failed', { error });
      res.status(500).json({
        error: 'Failed to create donation session'
      });
    }
  }
);

/**
 * POST /api/church/tokens
 * Purchase Bambi Tokens (one-time payment)
 */
router.post('/tokens',
  authLimiter,
  donationLimiter,
  validators.email,
  validators.positiveInt('quantity'),  // Number of 100-token bundles
  validateRequest,
  async (req, res) => {
    try {
      const { email, quantity } = req.body;
      const TOKENS_PER_BUNDLE = 100;
      const PRICE_PER_BUNDLE = 100;  // â‚¬1.00 in cents

      // Create or retrieve customer
      let customer;
      const existingCustomers = await stripe.customers.list({ email, limit: 1 });
      
      if (existingCustomers.data.length > 0) {
        customer = existingCustomers.data[0];
      } else {
        customer = await stripe.customers.create({
          email,
          metadata: {
            source: 'bambisleep-church',
            product: 'bambi-tokens'
          }
        });
      }

      // Create checkout session for one-time payment
      const session = await stripe.checkout.sessions.create({
        customer: customer.id,
        mode: 'payment',
        payment_method_types: ['card'],
        line_items: [{
          price_data: {
            currency: 'eur',
            product_data: {

            unit_amount: PRICE_PER_BUNDLE  // â‚¬1.00 in cents

        cancel_url: `${process.env.APP_URL}/tokens/cancel`,

          metadata: {
            product: 'bambi-tokens',

            total_tokens: quantity * TOKENS_PER_BUNDLE

        error: 'Failed to create token purchase session'

/**
 * GET /api/church/tokens/balance
 * Get user's token balance
 */
router.get('/tokens/balance',
  authLimiter,
  async (req, res) => {
    try {
      const userId = req.user?.id;  // Assume JWT authentication
      
      if (!userId) {
        return res.status(401).json({ error: 'Authentication required' });
      }

      const { pool } = await import('../services/database.js');
      const result = await pool.query(
        'SELECT balance, lifetime_earned, lifetime_spent, updated_at FROM token_balances WHERE user_id = $1',
        [userId]

          balance: 0,
          lifetimeEarned: 0,
          lifetimeSpent: 0,
          updatedAt: null
        });
      }

      const { balance, lifetime_earned, lifetime_spent, updated_at } = result.rows[0];

      res.json({
        balance,
        lifetimeEarned: lifetime_earned,

        error: 'Failed to retrieve token balance'

  }
```
